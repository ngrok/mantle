A date field component that allows users to enter and edit date.

Use `mode="single"` to let a user select a single date.

<Example>
	<SingleCalendarExample />
</Example>

```tsx
import { Calendar } from "@ngrok/mantle/calendar";
import { useState } from "react";

function Example() {
	const [date, setDate] = useState<Date | undefined>(undefined);

	return (
		<Calendar
			mode="single"
			selected={date}
			onSelect={setDate}
			className="border-card rounded-md border p-2 shadow-md"
		/>
	);
}
```

## Popover

Compose the `Calendar` inside a `Popover` to reveal it from a trigger.

<Example>
	<PopoverCalendarExample />
</Example>

```tsx
import { IconButton } from "@ngrok/mantle/button";
import { Calendar } from "@ngrok/mantle/calendar";
import { Popover } from "@ngrok/mantle/popover";
import { CalendarIcon } from "@phosphor-icons/react/Calendar";
import { useState } from "react";

function Example() {
	const [date, setDate] = useState<Date | undefined>(undefined);

	return (
		<Popover.Root>
			<Popover.Trigger asChild>
				<IconButton
					type="button"
					appearance="outlined"
					intent="neutral"
					icon={<CalendarIcon />}
					label="Open calendar popover"
				/>
			</Popover.Trigger>
			<Popover.Content>
				<Calendar mode="single" onSelect={(value) => setDate(value)} selected={date} />
			</Popover.Content>
		</Popover.Root>
	);
}
```

## Range

Use `mode="range"` to let a user select a start and end date.

<Example>
	<RangeCalendarExample />
</Example>

```tsx
import { Calendar, type DateRange } from "@ngrok/mantle/calendar";
import { useState } from "react";

function Example() {
	const [date, setDate] = useState<DateRange | undefined>({
		from: undefined,
		to: undefined,
	});

	return (
		<Calendar
			mode="range"
			defaultMonth={date?.from}
			selected={date}
			onSelect={setDate}
			className="border-card rounded-md border p-2 shadow-md"
		/>
	);
}
```

## Hidden Days

Pass an array of dates to `hidden` to remove them from the calendar.

<Example>
	<HiddenDaysCalendarExample />
</Example>

```tsx
import { Calendar, type DateRange } from "@ngrok/mantle/calendar";
import { useState } from "react";

function Example() {
	const [date, setDate] = useState<DateRange | undefined>({
		from: undefined,
		to: undefined,
	});
	const hiddenDays = [new Date(2024, 4, 6), new Date(2024, 4, 20), new Date(2024, 4, 11)];

	return (
		<Calendar
			mode="range"
			defaultMonth={hiddenDays[0]}
			selected={date}
			onSelect={setDate}
			hidden={hiddenDays}
			className="border-card rounded-md border p-2 shadow-md"
		/>
	);
}
```

## Disabled Days

Pass an array of dates to `disabled` to prevent them from being selected.

<Example>
	<DisabledDaysCalendarExample />
</Example>

```tsx
import { Calendar, type DateRange } from "@ngrok/mantle/calendar";
import { useState } from "react";

function Example() {
	const [date, setDate] = useState<DateRange | undefined>({
		from: undefined,
		to: undefined,
	});
	const disabledDays = [new Date(2024, 4, 3), new Date(2024, 4, 13), new Date(2024, 4, 14)];

	return (
		<Calendar
			mode="range"
			defaultMonth={disabledDays[0]}
			selected={date}
			onSelect={setDate}
			disabled={disabledDays}
			className="border-card rounded-md border p-2 shadow-md"
		/>
	);
}
```

## API Reference

The `Calendar` is built on top of [React DayPicker](https://react-day-picker.js.org/).

<PageHeader id="sidebar">Sidebar</PageHeader>

A composable, collapsible app-navigation sidebar. `Sidebar.Root` owns the state and renders no DOM;
`Sidebar.Nav` renders the panel — inline collapse-to-hidden on desktop, a left-side [Sheet](/components/overlays/sheet)
below the root's `mobileBreakpoint`; and `Sidebar.Trigger` toggles it from anywhere under the root, which is
what keeps the sidebar decoupled from your app shell: place the trigger in an
[App Layout](/layouts/app-layout) header without the shell knowing the sidebar exists.

The sidebar is routing-agnostic. Compose router links with `asChild` on `Sidebar.ItemButton` and drive
`current` from your router's location. Resize the window below `lg` to see the mobile sheet presentation, use
the trigger in the header to collapse and expand the panel, and toggle the notice to see the shell's
full-window [`AppLayout.Notice`](/layouts/app-layout) strip push everything — sidebar included — down.

```tsx
import { AppLayout } from "@ngrok/mantle/app-layout";
import { Breadcrumb } from "@ngrok/mantle/breadcrumb";
import { Button } from "@ngrok/mantle/button";
import { DropdownMenu } from "@ngrok/mantle/dropdown-menu";
import { Sidebar, useSidebar } from "@ngrok/mantle/sidebar";
import { ArrowsClockwiseIcon } from "@phosphor-icons/react/ArrowsClockwise";
import { CaretDownIcon } from "@phosphor-icons/react/CaretDown";
import { GlobeHemisphereWestIcon } from "@phosphor-icons/react/GlobeHemisphereWest";
import { GraphIcon } from "@phosphor-icons/react/Graph";
import { TerminalWindowIcon } from "@phosphor-icons/react/TerminalWindow";
import { WarningCircleIcon } from "@phosphor-icons/react/WarningCircle";
import { useState } from "react";

const navSections = [
	{
		title: undefined,
		items: [
			{ label: "Endpoints", icon: <GraphIcon />, path: "/endpoints" },
			{ label: "Agents", icon: <TerminalWindowIcon />, path: "/agents" },
		],
	},
	{
		title: "Traffic",
		items: [{ label: "Domains", icon: <GlobeHemisphereWestIcon />, path: "/domains" }],
	},
];

const accounts = [
	{ id: "acc_acme", name: "Acme Corp" },
	{ id: "acc_atlas", name: "Atlas Industries" },
];

function ProductNav({
	pathname,
	onNavigate,
}: {
	pathname: string;
	onNavigate: (path: string) => void;
}) {
	// Inside Sidebar.Root, useSidebar closes the mobile sheet on navigation.
	const { setOpenMobile } = useSidebar();

	return (
		<>
			{navSections.map((section) => (
				<Sidebar.Group key={section.title ?? "top-level"}>
					{section.title != null && <Sidebar.GroupLabel>{section.title}</Sidebar.GroupLabel>}
					<Sidebar.List>
						{section.items.map((item) => (
							<Sidebar.Item key={item.path}>
								<Sidebar.ItemButton asChild current={pathname === item.path}>
									<a
										href={item.path}
										onClick={(event) => {
											event.preventDefault();
											onNavigate(item.path);
											setOpenMobile(false);
										}}
									>
										{item.icon}
										{item.label}
									</a>
								</Sidebar.ItemButton>
							</Sidebar.Item>
						))}
					</Sidebar.List>
				</Sidebar.Group>
			))}
		</>
	);
}

export function AppShell() {
	const [pathname, setPathname] = useState("/endpoints");
	const [accountId, setAccountId] = useState("acc_acme");
	const [showNotice, setShowNotice] = useState(false);
	const account = accounts.find((candidate) => candidate.id === accountId) ?? accounts[0];

	return (
		<Sidebar.Root>
			<AppLayout.Root className="fixed inset-0">
				<AppLayout.Notice>
					{showNotice && (
						<div className="text-on-filled flex items-center gap-2 bg-red-600 px-4 py-1 text-xs">
							<WarningCircleIcon weight="fill" className="shrink-0" />
							You are impersonating jane@example.com in read-only mode.
						</div>
					)}
				</AppLayout.Notice>
				<AppLayout.Body>
					<Sidebar.Nav aria-label="Main">
						<Sidebar.Header>
							<DropdownMenu.Root>
								<DropdownMenu.Trigger asChild>
									<Sidebar.SwitcherButton>
										<span className="text-muted [&>svg]:size-5 [&>svg]:shrink-0">
											<GlobeHemisphereWestIcon />
										</span>
										<span className="text-strong min-w-0 flex-1 truncate text-base">
											Universal Gateway
										</span>
										<CaretDownIcon className="text-muted size-4 shrink-0" />
									</Sidebar.SwitcherButton>
								</DropdownMenu.Trigger>
								<DropdownMenu.Content align="start">{/* product switching */}</DropdownMenu.Content>
							</DropdownMenu.Root>
						</Sidebar.Header>

						<Sidebar.Body>
							<ProductNav pathname={pathname} onNavigate={setPathname} />
						</Sidebar.Body>

						<Sidebar.Footer>
							<Sidebar.Separator />
							<DropdownMenu.Root>
								<DropdownMenu.Trigger asChild>
									<Sidebar.SwitcherButton>
										<Sidebar.AccountAvatar accountId={account.id} accountName={account.name} />
										<span className="text-strong min-w-0 flex-1 truncate text-xs font-medium">
											{account.name}
										</span>
										<Sidebar.UserAvatar alt="Jane Doe" />
									</Sidebar.SwitcherButton>
								</DropdownMenu.Trigger>
								<DropdownMenu.Content align="start" side="top" className="min-w-56">
									<DropdownMenu.Sub>
										<DropdownMenu.SubTrigger className="gap-2">
											<ArrowsClockwiseIcon className="text-muted" />
											Switch accounts
										</DropdownMenu.SubTrigger>
										<DropdownMenu.SubContent>
											<Sidebar.SwitchAccountsRadioGroup
												accounts={accounts}
												value={accountId}
												onValueChange={setAccountId}
											/>
										</DropdownMenu.SubContent>
									</DropdownMenu.Sub>
								</DropdownMenu.Content>
							</DropdownMenu.Root>
						</Sidebar.Footer>
					</Sidebar.Nav>

					<AppLayout.Inset>
						<AppLayout.Content>
							<AppLayout.Header>
								<Sidebar.Trigger />
								<Breadcrumb.Root>
									<Breadcrumb.List>
										<Breadcrumb.Item>
											<Breadcrumb.Link href="/">Universal Gateway</Breadcrumb.Link>
										</Breadcrumb.Item>
										<Breadcrumb.Separator />
										<Breadcrumb.Item>
											<Breadcrumb.Page>Endpoints</Breadcrumb.Page>
										</Breadcrumb.Item>
									</Breadcrumb.List>
								</Breadcrumb.Root>
								<Button
									type="button"
									appearance="outlined"
									intent="neutral"
									className="ml-auto"
									size="sm"
									onClick={() => setShowNotice((current) => !current)}
								>
									Toggle notice
								</Button>
							</AppLayout.Header>
							{/* your routed page content */}
						</AppLayout.Content>
					</AppLayout.Inset>
				</AppLayout.Body>
			</AppLayout.Root>
		</Sidebar.Root>
	);
}
```

## Composition

Compose the parts of a `Sidebar` together to build your own:

```text showLineNumbers=false
Sidebar.Root
├── Sidebar.Nav
│   ├── Sidebar.Header
│   │   └── Sidebar.SwitcherButton
│   ├── Sidebar.Body
│   │   └── Sidebar.Group
│   │       ├── Sidebar.GroupLabel
│   │       └── Sidebar.List
│   │           └── Sidebar.Item
│   │               └── Sidebar.ItemButton
│   ├── Sidebar.Separator
│   └── Sidebar.Footer
│       └── Sidebar.SwitcherButton
│           ├── Sidebar.AccountAvatar
│           └── Sidebar.UserAvatar
└── Sidebar.Trigger
```

`Sidebar.Root` renders no DOM — it only provides state, so `Sidebar.Trigger` can live anywhere under it,
including inside layout components that know nothing about the sidebar. Render exactly one `Sidebar.Nav` per
`Sidebar.Root`; nested roots shadow the outer sidebar for everything below them.
`Sidebar.SwitchAccountsRadioGroup` is not part of the tree above — it renders inside a
[DropdownMenu](/components/overlays/dropdown-menu) submenu, wherever you compose that menu.

## Navigation anatomy

The navigation parts are plain styled elements: `Sidebar.Group` pairs an optional muted `Sidebar.GroupLabel`
with a `Sidebar.List` of rows, and links the two with `aria-labelledby` automatically so each list is
announced with its group's name. `Sidebar.ItemButton` renders a leading icon slot and a truncating label, and
`current` sets both `aria-current="page"` and the `data-current` visual state.

```tsx
	return (

				<Sidebar.GroupLabel>Network</Sidebar.GroupLabel>
				<Sidebar.List>
					<Sidebar.Item>
						<Sidebar.ItemButton asChild current={pathname === "/domains"}>
							<a
								href="/domains"
								onClick={(event) => {
									event.preventDefault();
									setPathname("/domains");
								}}
							>
								<GlobeHemisphereWestIcon />
								Domains
							</a>
						</Sidebar.ItemButton>
					</Sidebar.Item>
					<Sidebar.Item>
						<Sidebar.ItemButton asChild current={pathname === "/tcp-addresses"}>
							<a
								href="/tcp-addresses"
								onClick={(event) => {

								}}

					</Sidebar.Item>
```

## Persisting the collapsed state

`Sidebar.Root` supports both uncontrolled (`defaultOpen`) and controlled (`open` + `onOpenChange`) desktop
state, and the mobile sheet mirrors that with `openMobile` + `onOpenMobileChange`. To persist the collapsed
state across visits, use the **fully controlled** form with
[`useLocalStorage`](/hooks#uselocalstorage):

> [!IMPORTANT]
> In SSR apps, `defaultOpen` + storage does not work: the uncontrolled state initializes exactly once at the
> hydration render, before the stored value is available. Controlled `open` is the reliable form.

This recipe is SSR-safe: [`useLocalStorage`](/hooks#uselocalstorage) is `useSyncExternalStore`-backed with a
server snapshot, so the server render and the hydration render agree (no hydration mismatch), and the
sidebar deliberately suppresses its collapse transition until after hydration, so a persisted-collapsed
sidebar **snaps** shut instead of animating on page load. What no localStorage recipe can do is affect the
server-rendered HTML itself — a persisted-collapsed sidebar paints expanded for one frame before the
post-hydration correction. If that frame matters, persist to a **cookie** instead: read it in your server
loader and pass it as `open`/`defaultOpen`, the same way the theme system solves FOUC server-side.

<Example className="p-4 md:p-4">
	<SidebarPersistenceDemo />
</Example>

```tsx
import { AppLayout } from "@ngrok/mantle/app-layout";
import { useLocalStorage } from "@ngrok/mantle/hooks";
import { Sidebar } from "@ngrok/mantle/sidebar";
import { GraphIcon } from "@phosphor-icons/react/Graph";

export function PersistedShell() {
	const [storedState, setStoredState] = useLocalStorage("sidebar-state", "expanded");

	return (
		<Sidebar.Root
			open={storedState !== "collapsed"}
			onOpenChange={(open) => setStoredState(open ? "expanded" : "collapsed")}
		>
			<AppLayout.Root className="fixed inset-0">
				<AppLayout.Body>
					<Sidebar.Nav aria-label="Main">
						<Sidebar.Body>
							<Sidebar.Group>
								<Sidebar.GroupLabel>Persisted</Sidebar.GroupLabel>
								<Sidebar.List>
									<Sidebar.Item>
										<Sidebar.ItemButton current>
											<GraphIcon />
											Endpoints
										</Sidebar.ItemButton>
									</Sidebar.Item>
								</Sidebar.List>
							</Sidebar.Group>
						</Sidebar.Body>
					</Sidebar.Nav>
					<AppLayout.Inset>
						<AppLayout.Content>
							<AppLayout.Header>
								<Sidebar.Trigger />
							</AppLayout.Header>
							{/* your routed page content */}
						</AppLayout.Content>
					</AppLayout.Inset>
				</AppLayout.Body>
			</AppLayout.Root>
		</Sidebar.Root>
	);
}
```

## Polymorphism

`Sidebar.Header`, `Sidebar.Body`, `Sidebar.Footer`, `Sidebar.Group`, `Sidebar.GroupLabel`, `Sidebar.List`,
`Sidebar.Item`, `Sidebar.ItemButton`, and `Sidebar.SwitcherButton` accept an `asChild` prop. When `true`, the
part renders its single child instead of its default element, forwarding all class names, `data-*`
attributes, and the ref onto that child. The most common uses are router links on `Sidebar.ItemButton` (shown
throughout this page) and composing `Sidebar.SwitcherButton` with `DropdownMenu.Trigger asChild` or
`Dialog.Trigger asChild`.

`Sidebar.Nav` and `Sidebar.Trigger` deliberately do not support `asChild`: the nav renders a fixed
two-element collapse structure (and a `Sheet` on mobile), and the trigger renders a fully-wired
`IconButton` — build custom triggers with `useSidebar` instead. `Sidebar.AccountAvatar`,
`Sidebar.UserAvatar`, and `Sidebar.SwitchAccountsRadioGroup` are prop-driven leaves.

## Accessibility

- `Sidebar.Nav` renders a `<nav>` landmark named `Main` by default; override `aria-label` when your app has
  multiple navigation landmarks (breadcrumbs, pagination) or for localization.
- `Sidebar.Trigger` follows the disclosure pattern: `aria-expanded` reflects the state and `aria-controls`
  references the nav. Keep a trigger visible at every breakpoint where the sidebar can collapse — a collapsed
  sidebar with no visible trigger is unrecoverable.
- `⌘B` / `Ctrl+B` toggles the sidebar (opt out with `keyboardShortcut={false}` on `Sidebar.Root`, e.g. when
  the app embeds a rich-text editor where `⌘B` means bold).
- When the shell owns the document, pair the sidebar with a `SkipToMainLink` and a `Main`-composed content
  card so keyboard users can jump past the sidebar's tab gauntlet — see
  [App Layout → Landmarks](/layouts/app-layout#landmarks).
- Collapsed content is `visibility: hidden`: out of the tab order and the accessibility tree, not just
  visually hidden.
- `Sidebar.ItemButton` `current` renders `aria-current="page"`; `Sidebar.GroupLabel` is linked to its
  `Sidebar.List` via `aria-labelledby`.
- The mobile sheet is a modal dialog named "Sidebar" with focus trapping and `Escape` dismissal, courtesy of
  [Sheet](/components/overlays/sheet).
- The collapse/expand transitions respect `prefers-reduced-motion`.

## API Reference

### Sidebar.Root

The state owner. Renders no DOM.

| Prop                  | Type                      | Default | Description                                                                                       |
| --------------------- | ------------------------- | ------- | ------------------------------------------------------------------------------------------------- |
| `defaultOpen?`        | `boolean`                 | `true`  | Initial desktop expanded state for the uncontrolled case.                                         |
| `open?`               | `boolean`                 | —       | Controlled desktop expanded state. Pair with `onOpenChange`.                                      |
| `onOpenChange?`       | `(open: boolean) => void` | —       | Called with the next desktop expanded state.                                                      |
| `openMobile?`         | `boolean`                 | —       | Controlled mobile sheet open state. Pair with `onOpenMobileChange`.                               |
| `onOpenMobileChange?` | `(open: boolean) => void` | —       | Called with the next mobile sheet open state.                                                     |
| `mobileBreakpoint?`   | `"sm" \| "md" \| "lg"`    | `"lg"`  | Breakpoint below which `Sidebar.Nav` renders as a mobile `Sheet`.                                 |
| `keyboardShortcut?`   | `boolean`                 | `true`  | Toggle with `⌘B` / `Ctrl+B`. Exact-modifier match (`⌘⇧B` passes through); set `false` to opt out. |

### Sidebar.Nav

The panel. All props from `div` plus the panel contract: `className`, `style`, `ref`, and rest props land on
the panel surface in both presentations (the desktop collapse container and the mobile `Sheet.Content`), and
`aria-label` (default `"Main"`) is forwarded to the inner `<nav>` landmark. Emits
`data-state="expanded" | "collapsed"` on desktop and `data-mobile` on the mobile sheet.

| CSS Variable             | Default | Description                                                             |
| ------------------------ | ------- | ----------------------------------------------------------------------- |
| `--sidebar-width`        | `16rem` | Expanded desktop width. Set via `Sidebar.Nav`'s `className` or `style`. |
| `--sidebar-width-mobile` | `18rem` | Mobile sheet width. Set via `Sidebar.Nav`'s `className` or `style`.     |

### Sidebar.Trigger

An [IconButton](/components/actions/icon-button) wired to toggle the sidebar. All `IconButton` props except
`asChild` and `children`, with sidebar defaults:

| Prop          | Type                                | Default                 | Description                                 |
| ------------- | ----------------------------------- | ----------------------- | ------------------------------------------- |
| `appearance?` | `"ghost" \| "outlined"`             | `"ghost"`               | The visual style of the trigger.            |
| `icon?`       | `ReactNode`                         | `<SidebarSimpleIcon />` | The icon rendered inside the trigger.       |
| `intent?`     | `"accent" \| "danger" \| "neutral"` | `"neutral"`             | The color intent of the trigger.            |
| `label?`      | `string`                            | `"Toggle Sidebar"`      | Accessible name; override for localization. |

### Sidebar.Header, Sidebar.Body, Sidebar.Footer

Structural containers inside `Sidebar.Nav`: a pinned header, the scrollable middle, and a pinned footer. All
props from `div` plus `asChild`.

### Sidebar.Group, Sidebar.GroupLabel, Sidebar.List, Sidebar.Item

The navigation grouping: `Group` (a `div`) pairs an optional `GroupLabel` (a `div` — pass `asChild` to render
a heading at a level you control) with a `List` (a `ul`) of `Item` rows (`li`). When a label is present the
group links it to the list via `aria-labelledby`. All props from their respective elements plus `asChild`.

### Sidebar.ItemButton

The interactive navigation row. All props from `button` plus:

| Prop       | Type      | Default | Description                                                            |
| ---------- | --------- | ------- | ---------------------------------------------------------------------- |
| `asChild?` | `boolean` | `false` | Render the single child (e.g. a router link) instead of a `<button>`.  |
| `current?` | `boolean` | —       | Marks the current page: sets `aria-current="page"` and `data-current`. |

### Sidebar.SwitcherButton

The styled switcher row for the header (app switcher) and footer (account/user row). A styled button only —
compose it with `DropdownMenu.Trigger asChild` or `Dialog.Trigger asChild`, which provide the open state
(`data-state="open"` styling responds automatically). All props from `button` plus `asChild`.

### Sidebar.Separator

An inset hairline between sidebar regions, aligned with the `px-3` content padding of `Sidebar.Body` and
`Sidebar.Footer` (it deliberately does not run edge to edge). All props from
[Separator](/components/structure/separator).

### Sidebar.AccountAvatar

A rounded-square account avatar. The swatch color derives deterministically from `accountId` (stable across
sessions and devices) and every swatch keeps at least 4.5:1 contrast under the white initials.

| Prop          | Type                  | Default | Description                                              |
| ------------- | --------------------- | ------- | -------------------------------------------------------- |
| `accountId`   | `string \| undefined` | —       | Stable id; seeds the deterministic swatch color.         |
| `accountName` | `string \| undefined` | —       | Display name; its first 1–2 letters become the initials. |

### Sidebar.UserAvatar

A circular user avatar with a silhouette fallback.

| Prop   | Type     | Default          | Description                                                   |
| ------ | -------- | ---------------- | ------------------------------------------------------------- |
| `src?` | `string` | —                | Profile picture URL; falls back to a silhouette when omitted. |
| `alt?` | `string` | `"Your account"` | Accessible label for the avatar.                              |

### Sidebar.SwitchAccountsRadioGroup

A pre-styled radio group of accounts for a "Switch accounts" `DropdownMenu` submenu. All props from Radix
`DropdownMenu.RadioGroup` (`value`, `onValueChange`, …) plus:

| Prop       | Type                            | Default | Description                                        |
| ---------- | ------------------------------- | ------- | -------------------------------------------------- |
| `accounts` | `ReadonlyArray<SidebarAccount>` | —       | The accounts to render, `{ id, name, trailing? }`. |

### useSidebar

A hook returning the nearest `Sidebar.Root` state (`SidebarState`) — throws outside a root. Use it for custom
triggers, close-on-navigate, or a keyboard shortcut:

| Field              | Type                      | Description                                                  |
| ------------------ | ------------------------- | ------------------------------------------------------------ |
| `open`             | `boolean`                 | Desktop expanded state.                                      |
| `setOpen`          | `(open: boolean) => void` | Set the desktop expanded state.                              |
| `openMobile`       | `boolean`                 | Mobile sheet open state.                                     |
| `setOpenMobile`    | `(open: boolean) => void` | Set the mobile sheet state (e.g. close on navigation).       |
| `isMobile`         | `boolean`                 | Whether the viewport is below `mobileBreakpoint`.            |
| `mobileBreakpoint` | `"sm" \| "md" \| "lg"`    | The configured mobile breakpoint.                            |
| `toggle`           | `() => void`              | Toggle whichever presentation is active.                     |
| `navId`            | `string`                  | The nav landmark's id, for custom triggers' `aria-controls`. |

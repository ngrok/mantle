export const accounts = Array.from({ length: 14 }, (_, index) => {
	const names = [
		"Cody Price",
		"cody@ngrok.com",
		"corp@ngrok.com",
		"dogfood@ngrok.com",
		"ngrok eng (dev)",
		"acme corp",
		"globex",
	];
	const plans = ["All Subscription", "Enterprise Subscription", "Free"];
	const name = names[index % names.length];
	return {
		id: `account-${index}`,
		name: index < names.length ? name : `${name} ${Math.floor(index / names.length) + 1}`,
		plan: plans[index % plans.length],
		memberCount: (index * 7 + 1) % 120,
	};
});

export function ScrollableListExample() {
	const [activeId, setActiveId] = useState("account-2");
	return (
		<div className="w-full max-w-md">
			<ScrollableList.Viewport aria-label="Your accounts" className="max-h-80">
				{accounts.map((account) => (
					<ScrollableList.Item
						key={account.id}
						selected={account.id === activeId}
						onClick={() => setActiveId(account.id)}
					>
						<ScrollableList.ItemTitle>{account.name}</ScrollableList.ItemTitle>
						<ScrollableList.ItemDescription>
							{account.plan} · {account.memberCount} members
						</ScrollableList.ItemDescription>
					</ScrollableList.Item>
				))}
			</ScrollableList.Viewport>
		</div>
	);
}

<PageHeader id="scrollable-list">Scrollable List</PageHeader>

A scrollable list of clickable rows — the action / navigation counterpart to [`SelectableList`](/components/selectable-list). Compose `ScrollableList.Item` rows directly inside a `ScrollableList.Viewport`; each is a `<button>` (with `onClick`) or your own element via `asChild` (e.g. an `<a>`), with an optional `selected` highlight. It shares `SelectableList`'s look — styled like the [`MultiSelect`](/components/multi-select) popover, with rows that highlight on hover / selection as an inset, rounded pill inside a bordered `bg-popover` container — and suits an account switcher or an SSO provider picker. It renders every row by default; for long lists, swap `Viewport` → [`VirtualViewport`](#virtualization) to window them.

<Example>
	<ScrollableListExample />
</Example>

## Composition

```text showLineNumbers=false
ScrollableList.Viewport        (or .VirtualViewport)
└── ScrollableList.Item
    ├── ScrollableList.ItemTitle
    └── ScrollableList.ItemDescription
```

- **`Viewport`** renders the rounded, scrollable `bg-popover` container and its `Item` children (all of them — non-virtualized). Reads each item's `selected` / `disabled` to style its row. **Bound its height** (`max-h-*`, `h-*`, or `min-h-0 flex-1` in a flex parent) so long lists scroll.
- **`VirtualViewport`** is the drop-in windowed variant — same `Item` children, only the visible slice mounted. See [Virtualization](#virtualization).
- **`Item`** is a clickable row — a `<button>` by default (`onClick`), or your own element via `asChild`. `selected` gives the tinted pill.
- **`ItemTitle` / `ItemDescription`** are optional styled spans for the emphasized title and de-emphasized sub-line.

## Virtualization

`ScrollableList.Viewport` renders every row — simplest, and right for the short lists most action/navigation menus are. For a long list (hundreds+ of rows), swap in `ScrollableList.VirtualViewport`: it windows the same `Item` children via [`@tanstack/react-virtual`](https://tanstack.com/virtual), mounting only what's visible. The call site is otherwise identical — same children, same props — and `@tanstack/react-virtual` is only pulled into your bundle when you use `VirtualViewport`.

<Example>
	<ScrollableListVirtualExample />
</Example>

```tsx
import { ScrollableList } from "@ngrok/mantle/scrollable-list";
import { useState } from "react";

// `VirtualViewport` is a drop-in for `Viewport` — same `Item` children, same props.
function Example() {
	const [activeId, setActiveId] = useState("account-2");

	return (
		<ScrollableList.VirtualViewport aria-label="Your accounts" className="max-h-80">
			{accounts.map((account) => (
				<ScrollableList.Item
					key={account.id}
					selected={account.id === activeId}
					onClick={() => setActiveId(account.id)}
				>
					<ScrollableList.ItemTitle>{account.name}</ScrollableList.ItemTitle>
					<ScrollableList.ItemDescription>
						{account.plan} · {account.memberCount} members
					</ScrollableList.ItemDescription>
				</ScrollableList.Item>
			))}
		</ScrollableList.VirtualViewport>
	);
}
```

> **Windowing trade-off.** Off-screen rows aren't in the DOM, so Tab and browser find-in-page only reach mounted rows — the right trade for a large enough list. Prefer `Viewport` until a list is actually long.

## Polymorphism

Render rows as links by passing `asChild` and an `<a>` — the row chrome (hover pill, focus ring) applies to your element, so clicking navigates. Rows compose freely: each one here is a [`MediaObject`](/components/media-object) with the provider icon on the left and the name over a de-emphasized owner on the right — an SSO identity-provider picker, as a list rather than a stack of cards.

## Accessibility

The list is a `role="list"` of `role="listitem"` rows (the tag is a `<div>` — per WAI-ARIA the role, not the tag, carries the semantics); each `Item` is a focusable `<button>` (or your `asChild` element) inside its row. It is a **non-selecting** list, not a selection widget — that's [`SelectableList`](/components/selectable-list)'s job. Provide an `aria-label` (or `aria-labelledby`) on the `Viewport` so the list has an accessible name. Disabled rows set the control's `disabled` (button) or `aria-disabled` (`asChild`) and are dimmed.

Keyboard support is the native tab order of the rendered rows. Under `VirtualViewport`, off-screen rows aren't reachable by Tab (see [Virtualization](#virtualization)) — the windowing trade-off for a long list.

## API Reference

### ScrollableList.Viewport

The non-virtualized container. Accepts all `<div>` props, plus:

| Prop       | Type        | Default | Description                                    |
| ---------- | ----------- | ------- | ---------------------------------------------- |
| `children` | `ReactNode` |         | `ScrollableList.Item` rows (all are rendered). |

Bound the height (e.g. `className="max-h-80"`) so long lists scroll.

### ScrollableList.VirtualViewport

The windowed container — same surface as `Viewport` (compose the same `Item` children), plus the virtualizer knobs:

| Prop                 | Type        | Default | Description                                                            |
| -------------------- | ----------- | ------- | ---------------------------------------------------------------------- |
| `children`           | `ReactNode` |         | `ScrollableList.Item` rows (only the visible window is mounted).       |
| `estimateRowHeight?` | `number`    | `44`    | Estimated row height (px) used to seed the virtualizer before measure. |
| `overscan?`          | `number`    | `8`     | Rows rendered beyond the visible window on each side.                  |

### ScrollableList.Item

A clickable row. Accepts all `<button>` props (minus `type`), plus:

| Prop        | Type      | Default | Description                                                                           |
| ----------- | --------- | ------- | ------------------------------------------------------------------------------------- |
| `selected?` | `boolean` | `false` | Marks the row selected/active (tinted pill). Omit for a plain action/navigation list. |
| `disabled?` | `boolean` | `false` | Disables the row (dimmed; `disabled` on the button, `aria-disabled` when `asChild`).  |
| `asChild?`  | `boolean` | `false` | Render the row as the provided child element (e.g. an `<a>`) instead of a `<button>`. |

### ScrollableList.ItemTitle / ScrollableList.ItemDescription

Styled `<span>`s for the title and de-emphasized sub-line inside an `Item`. Both accept all `<span>` props.
